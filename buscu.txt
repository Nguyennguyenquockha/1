local library = loadstring(game:HttpGetAsync("https://raw.githubusercontent.com/Nguyennguyenquockha/Kha-hub/refs/heads/main/Gui_script.txt"))()local Main = library:CreateWindow("script bản v1 testing | bởi sey_roblox ","Emerald")local tab = Main:CreateTab("Chính")tab:CreateButton("Aimbot", function()    loadstring(game:HttpGet('https://raw.githubusercontent.com/Nguyennguyenquockha/1/refs/heads/main/Ch.txt'))()    print("clicked")end)tab:CreateButton("Fly Gui V3", function()    loadstring(game:HttpGet("https://raw.githubusercontent.com/XNEOFF/FlyGuiV3/main/FlyGuiV3.txt"))()    print("clicked")end)local espEnabled = falselocal espFolder = Instance.new("Folder", game.CoreGui)espFolder.Name = "PlayerESP"function createESP(plr)    if plr == game.Players.LocalPlayer then return end    local function onCharacter(char)        local hrp = char:WaitForChild("HumanoidRootPart", 5)        if not hrp then return end        -- Tên        local nameBillboard = Instance.new("BillboardGui", espFolder)        nameBillboard.Adornee = hrp        nameBillboard.Size = UDim2.new(0, 100, 0, 30)        nameBillboard.StudsOffset = Vector3.new(0, 3, 0)        nameBillboard.AlwaysOnTop = true        local nameLabel = Instance.new("TextLabel", nameBillboard)        nameLabel.Size = UDim2.new(1, 0, 1, 0)        nameLabel.BackgroundTransparency = 1        nameLabel.Text = plr.Name        nameLabel.TextColor3 = Color3.fromRGB(255, 255, 255)        nameLabel.TextSize = 14        nameLabel.Font = Enum.Font.Gotham        -- Tracer        local tracer = Drawing.new("Line")        tracer.Thickness = 1.5        tracer.Color = Color3.new(1, 1, 1)        tracer.Transparency = 1        -- Aura        local aura = Instance.new("SelectionBox", espFolder)        aura.Adornee = hrp        aura.LineThickness = 0.05        aura.SurfaceColor3 = Color3.fromRGB(0, 255, 255)        aura.SurfaceTransparency = 0.7        aura.Color3 = Color3.fromRGB(0, 255, 255)        -- Update tracer        local conn        conn = game:GetService("RunService").RenderStepped:Connect(function()            if not plr.Character or not plr.Character:FindFirstChild("HumanoidRootPart") or not espEnabled then                tracer.Visible = false                conn:Disconnect()                return            end            local pos, onscreen = workspace.CurrentCamera:WorldToViewportPoint(hrp.Position)            if onscreen then                tracer.Visible = true                tracer.From = Vector2.new(workspace.CurrentCamera.ViewportSize.X / 2, workspace.CurrentCamera.ViewportSize.Y)                tracer.To = Vector2.new(pos.X, pos.Y)            else                tracer.Visible = false            end        end)    end    if plr.Character then        onCharacter(plr.Character)    end    plr.CharacterAdded:Connect(onCharacter)endfunction clearESP()    for _, v in pairs(espFolder:GetChildren()) do        v:Destroy()    end    for _, obj in pairs(Drawing.GetDrawings()) do        if obj.Type == "Line" then            obj:Remove()        end    endendtab:CreateToggle("ESP người chơi", function(a)    espEnabled = a    clearESP()    if espEnabled then        for _, plr in pairs(game.Players:GetPlayers()) do            createESP(plr)        end        game.Players.PlayerAdded:Connect(function(plr)            plr.CharacterAdded:Connect(function()                wait(1)                if espEnabled then createESP(plr) end            end)        end)    endend)tab:Show()
