-- Key system (24h)
local DataStoreKey = "KhaHubKeyTime"
local CorrectKey = "KHADEPTRAIVAILON100"

-- Save time key used
local function saveKeyTime()
	if isfile and writefile then
		writefile(DataStoreKey, tostring(os.time()))
	end
end

-- Load time key used
local function loadKeyTime()
	if isfile and readfile and isfile(DataStoreKey) then
		return tonumber(readfile(DataStoreKey))
	end
	return nil
end

-- Check Key
